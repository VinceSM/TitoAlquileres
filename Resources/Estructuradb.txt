Create Database alquileres;

USE alquileres;

-- Tabla Items
CREATE TABLE Items (
    id INT IDENTITY(1,1) PRIMARY KEY,
    nombreItem VARCHAR(100),
    marca VARCHAR(50),
    modelo VARCHAR(50),
    tarifaDia DOUBLE PRECISION,
    categoriaId INT,
    deletedAt DATETIME
);

-- Tabla Categorias
CREATE TABLE Categorias (
    id INT IDENTITY(1,1) PRIMARY KEY,
    nombre VARCHAR(50),
    deletedAt DATETIME
);

-- Tabla Transportes
CREATE TABLE Transportes (
    id INT IDENTITY(1,1) PRIMARY KEY,
    item_id INT UNIQUE,
    capacidadPasajeros INT,
    tipoCombustible VARCHAR(50),
    FOREIGN KEY (item_id) REFERENCES Items(id)
);

-- Tabla Electronicas
CREATE TABLE Electronicas (
    id INT IDENTITY(1,1) PRIMARY KEY,
    item_id INT UNIQUE,
    resolucionPantalla VARCHAR(50),
    almacenamientoGB INT,
    FOREIGN KEY (item_id) REFERENCES Items(id)
);

-- Tabla Electrodomesticos
CREATE TABLE Electrodomesticos (
    id INT IDENTITY(1,1) PRIMARY KEY,
    item_id INT UNIQUE,
    potenciaWatts INT,
    tipoElectrodomestico VARCHAR(50),
    FOREIGN KEY (item_id) REFERENCES Items(id)
);

-- Tabla Inmuebles
CREATE TABLE Inmuebles (
    id INT IDENTITY(1,1) PRIMARY KEY,
    item_id INT UNIQUE,
    metrosCuadrados INT,
    ubicacion VARCHAR(100),
    FOREIGN KEY (item_id) REFERENCES Items(id)
);

-- Tabla Indumentarias
CREATE TABLE Indumentarias (
    id INT IDENTITY(1,1) PRIMARY KEY,
    item_id INT UNIQUE,
    talla VARCHAR(10),
    material VARCHAR(50),
    FOREIGN KEY (item_id) REFERENCES Items(id)
);

-- Tabla Usuarios
CREATE TABLE Usuarios (
    id INT IDENTITY(1,1) PRIMARY KEY,
    nombre VARCHAR(100),
    dni INT UNIQUE,
    email VARCHAR(100),
    membresiaPremium BIT,
    deletedAt DATETIME
);

-- Tabla Alquileres
CREATE TABLE Alquileres (
    id INT IDENTITY(1,1) PRIMARY KEY,
    ItemID INT,
    UsuarioID INT,
    tiempoDias INT,
    fechaInicio DATETIME,
    fechaFin DATETIME,
    precioTotal DOUBLE PRECISION,
    tipoEstrategia VARCHAR(50),
    deletedAt DATETIME,
    FOREIGN KEY (ItemID) REFERENCES Items(id),
    FOREIGN KEY (UsuarioID) REFERENCES Usuarios(id)
);

-- Relaciones adicionales
ALTER TABLE Items
ADD CONSTRAINT FK_Items_Categorias FOREIGN KEY (categoriaId) REFERENCES Categorias(id);
